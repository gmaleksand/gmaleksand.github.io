  2. –í—ä—Ä—Ç—è—â –º–æ–º–µ–Ω—Ç / –º–æ–º–µ–Ω—Ç –Ω–∞ —Å–∏–ª–∞ ($\vec{\tau}, [\text{N}\cdot \text{m}]$) - –≤–µ–ª–∏—á–∏–Ω–∞, —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑–∏—Ä–∞—â–∞ –≤—ä—Ä—Ç–µ–ª–∏–≤–æ—Ç–æ –¥–≤–∏–∂–µ–Ω–∏–µ, –∫–æ–µ—Ç–æ –¥–∞–¥–µ–Ω–∞ —Å–∏–ª–∞ –ø—Ä–∏–¥–∞–≤–∞ –Ω–∞ —Ç—è–ª–æ—Ç–æ, –≤—ä—Ä—Ö—É –∫–æ–µ—Ç–æ –≤—ä–∑–¥–µ–π—Å—Ç–≤–∞

    $$\vec{\tau} \overset{\text{def}}{=} \vec{r}\times \vec{F}$$

    –∞) —Ä–∞–º–æ –Ω–∞ —Å–∏–ª–∞—Ç–∞ ($l$) - —Ä–∞–∑—Å—Ç–æ—è–Ω–∏–µ—Ç–æ –º–µ–∂–¥—É –æ—Å—Ç–∞ –Ω–∞ –≤—ä—Ä—Ç–µ–Ω–µ –∏ –ø—Ä–∞–≤–∞—Ç–∞, –ø–æ –∫–æ—è—Ç–æ –¥–µ–π—Å—Ç–≤–∞ —Å–∏–ª–∞—Ç–∞

    –±) –≥–æ–ª–µ–º–∏–Ω–∞

    $$\tau = Fl$$

    • —Å–∏–ª–∞, –∫–æ—è—Ç–æ –¥–µ–π—Å—Ç–≤–∞ —Ç–∞–∫–∞, —á–µ –¥–∞ –∑–∞–≤—ä—Ä—Ç–∏ —Ç—è–ª–æ—Ç–æ –ø–æ –ø–æ—Å–æ–∫–∞, –æ–±—Ä–∞—Ç–Ω–∞ –Ω–∞ —á–∞—Å–æ–≤–Ω–∏–∫–æ–≤–∞—Ç–∞ —Å—Ç—Ä–µ–ª–∫–∞, –∏–º–∞ –ø–æ–ª–æ–∂–∏—Ç–µ–ª–µ–Ω –≤—ä—Ä—Ç—è—â –º–æ–º–µ–Ω—Ç
    • —Å–∏–ª–∞, –∫–æ—è—Ç–æ –¥–µ–π—Å—Ç–≤–∞ —Ç–∞–∫–∞, —á–µ –¥–∞ –∑–∞–≤—ä—Ä—Ç–∏ —Ç—è–ª–æ—Ç–æ –ø–æ –ø–æ—Å–æ–∫–∞ –Ω–∞ —á–∞—Å–æ–≤–Ω–∏–∫–æ–≤–∞—Ç–∞ —Å—Ç—Ä–µ–ª–∫–∞, –∏–º–∞ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–µ–Ω –≤—ä—Ä—Ç—è—â –º–æ–º–µ–Ω—Ç

    –≤) –ø–æ—Å–æ–∫–∞ - –ø–∞–ª–µ—Ü—ä—Ç –Ω–∞ –¥—è—Å–Ω–∞—Ç–∞ —Ä—ä–∫–∞ —Å–æ—á–∏ –ø–æ –ø–æ—Å–æ–∫–∞ –Ω–∞ –≤–µ–∫—Ç–æ—Ä–∞ –Ω–∞ –≤—ä—Ä—Ç—è—â–∏—è –º–æ–º–µ–Ω—Ç, –∫–æ–≥–∞—Ç–æ –æ—Å—Ç–∞–Ω–∞–ª–∏—Ç–µ –ø—Ä—ä—Å—Ç–∏ —Å–µ –Ω–∞—Å–æ—á–∞—Ç –ø–æ –ø–æ—Å–æ–∫–∞ –Ω–∞ $\vec{r}$ –∏ —Å–µ —Å–≤–∏—è—Ç –≤ –ø–æ—Å–æ–∫–∞ –Ω–∞ —Å–∏–ª–∞—Ç–∞

    –≥) –µ—Ñ–µ–∫—Ç –≤—ä—Ä—Ö—É –º–æ–º–µ–Ω—Ç–∞ –Ω–∞ –∏–º–ø—É–ª—Å–∞ - –≤—ä—Ä—Ç—è—â–∏—è—Ç –º–æ–º–µ–Ω—Ç –µ –ø—Ä–æ–º—è–Ω–∞—Ç–∞ –≤ –º–æ–º–µ–Ω—Ç–∞ –Ω–∞ –∏–º–ø—É–ª—Å–∞

    $$\frac{d\vec{L}}{dt} = \vec{\tau}$$ –î–æ–∫–∞–∑–∞—Ç–µ–ª—Å—Ç–≤–æ: $$\frac{d\vec L}{dt} = \frac{d (m\vec r \times \vec v)}{dt} = m \frac{d\vec r}{dt} \times \vec v + m \vec r \times \frac{d \vec v}{dt} = m \vec v \times \vec v + m \vec r \times \vec a = \vec r \times m\vec a$$

  3. –ó–∞–∫–æ–Ω –∑–∞ –∑–∞–ø–∞–∑–≤–∞–Ω–µ –Ω–∞ –º–æ–º–µ–Ω—Ç–∞ –Ω–∞ –∏–º–ø—É–ª—Å–∞ - –º–æ–º–µ–Ω—Ç—ä—Ç –Ω–∞ –∏–º–ø—É–ª—Å–∞ –≤ –∑–∞—Ç–≤–æ—Ä–µ–Ω–∏ —Å–∏—Å—Ç–µ–º–∏ –æ—Å—Ç–∞–≤–∞ –ø–æ—Å—Ç–æ—è–Ω–µ–Ω –≤—ä–≤ –≤—Ä–µ–º–µ—Ç–æ

    –î–æ–∫–∞–∑–∞—Ç–µ–ª—Å—Ç–≤–æ:

    –ü—Ä–æ–º—è–Ω–∞—Ç–∞ –Ω–∞ –º–æ–º–µ–Ω—Ç–∞ –Ω–∞ –∏–º–ø—É–ª—Å–∞ –Ω–∞ –µ–¥–Ω–æ —Ç—è–ª–æ –µ —Å–±–æ—Ä–∞ –æ—Ç –≤—ä—Ä—Ç—è—â–∏—Ç–µ –º–æ–º–µ–Ω—Ç–∏, –∫–æ–∏—Ç–æ –º—É –¥–µ–π—Å—Ç–≤–∞—Ç: $$\frac{d \vec L_1}{dt} = \vec r_1 \times \vec F_{21} + ‚Ä¶ \vec r_1 \times \vec F_{n1} $$ $$\frac{d \vec L_2}{dt} = \vec r_2 \times \vec F_{12} + ‚Ä¶ \vec r_2 \times \vec F_{n1} $$ $$\frac{d \vec L_n}{dt} = \vec r_n \times \vec F_{1n} + \vec r_n \times \vec F_{2n} + ‚Ä¶$$ –ö–∞—Ç–æ —Å—ä–±–µ—Ä–µ–º –≤—Å–∏—á–∫–∏ —É—Ä–∞–≤–Ω–µ–Ω–∏—è, —â–µ –∏–º–∞–º–µ –¥–≤–æ–π–∫–∏ –≤—ä—Ä—Ç—è—â–∏ –º–æ–º–µ–Ω—Ç–∏ –æ—Ç –≤–∏–¥–∞ $$\vec r_i \times \vec F_{ji} + \vec r_j \times \vec F_{ij} = \vec r_i \times \vec F_{ji} - \vec r_j \times \vec F_{ji} = (\vec r_i - \vec r_j) \times \vec F_{ji} = \vec 0,$$ —Ç—ä–π –∫–∞—Ç–æ $\vec F_{ji}$ –µ –∫–æ–ª–∏–Ω–µ–∞—Ä–Ω–∞ —Å $\vec r_i - \vec r_j$. –ö–∞—Ç–æ —Å—ä–∫—Ä–∞—Ç–∏–º –≤—Å–∏—á–∫–∏ –¥–≤–æ–π–∫–∏ –≤—ä—Ä—Ç—è—â–∏ –º–æ–º–µ–Ω—Ç–∏, $$\frac{d\vec{L}}{dt}= \vec0$$

    –∞) –ø—Ä–∏ –æ—Ç–≤–æ—Ä–µ–Ω–∏ —Å–∏—Å—Ç–µ–º–∏ - –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—Ç–∞ –Ω–∞ –º–æ–º–µ–Ω—Ç–∞ –Ω–∞ –∏–º–ø—É–ª—Å–∞ –µ —Ä–∞–≤–Ω–∞ –Ω–∞ —Å–±–æ—Ä–∞ –æ—Ç –≤—Å–∏—á–∫–∏ –≤—ä–Ω—à–Ω–∏ –≤—ä—Ä—Ç—è—â–∏ –º–æ–º–µ–Ω—Ç–∏
